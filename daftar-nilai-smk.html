🔧 Panduan Setup Google Sheets

Langkah 1. Buka spreadsheet → klik Extensions → Apps Script.

Langkah 2. Hapus kode lama, paste kode ini:

const SHEET_ID = "1ncWuZxFDaDqNFYNn2Co052JvGYYBdSn-nzrcBhi4jZ8"; const SHEET_NAME = "Sheet1"; function doPost(e) { try { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName(SHEET_NAME); const rows = JSON.parse(e.postData.contents); const last = sh.getLastRow(); if (last > 1) sh.getRange(2,1,last-1,15).clearContent(); rows.forEach((r,i) => { sh.getRange(i+2,1,1,15).setValues([[ r.no,r.nama,r.nis,r.nisn,r.jk, r.f1,r.f2,r.f3,r.avgF, r.s1,r.s2,r.s3,r.avgS, r.sas,r.rapor ]]); }); return ContentService .createTextOutput(JSON.stringify({status:"ok",rows:rows.length})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { if (e.parameter.data) { // Terima data via GET (fallback CORS) try { const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName(SHEET_NAME); const rows = JSON.parse(decodeURIComponent(e.parameter.data)); const last = sh.getLastRow(); if (last > 1) sh.getRange(2,1,last-1,15).clearContent(); rows.forEach((r,i) => { sh.getRange(i+2,1,1,15).setValues([[ r.no,r.nama,r.nis,r.nisn,r.jk, r.f1,r.f2,r.f3,r.avgF, r.s1,r.s2,r.s3,r.avgS, r.sas,r.rapor ]]); }); return ContentService .createTextOutput(JSON.stringify({status:"ok",rows:rows.length})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } // Normal GET = baca data const ss = SpreadsheetApp.openById(SHEET_ID); const sh = ss.getSheetByName(SHEET_NAME); const data = sh.getDataRange().getValues().slice(1) .filter(r=>r[1]).map(r=>({ no:r[0],nama:r[1],nis:r[2],nisn:r[3],jk:r[4], f1:r[5],f2:r[6],f3:r[7],avgF:r[8], s1:r[9],s2:r[10],s3:r[11],avgS:r[12], sas:r[13],rapor:r[14] })); return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }

Langkah 3. Deploy → New Deployment → Web App. Set Execute as: Me & Who has access: Anyone. Copy URL-nya.

Langkah 4. Paste URL ke konstanta GAS_URL di baris paling atas <script>. Simpan file HTML.

💡 Catatan CORS: Aplikasi ini otomatis mencoba POST dulu, lalu fallback ke GET jika ada blokir CORS. Kedua cara sudah didukung oleh kode Apps Script di atas.
